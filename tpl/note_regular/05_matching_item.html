匹配子项

匹配子项 ： 小括号 () (另一个意思，分组操作)
var str = "2016-6-18";
var reg = /(\d+)(-)/g;
str = str.replace( reg , function(string , string01 , string02){
  // string----正则整体 , string01----第一个子项 , string02----第二个子项
  return string01 + ".";
});
console.log(str); //2016.6.18
str = "abc"
reg = /(a)(b)(c)/;
console.log(str.match(reg)); //[abc , a , b , c] --- 子项会被自动的push到匹配成功的数组后(仅适用于非全局匹配时)

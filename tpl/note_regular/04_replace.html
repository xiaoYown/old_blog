replace

正则去匹配字符串 ， 匹配成功的字符被替换成新的字符
语法 ： string.replace( reg , string | callback)
example:
var str = "agghdsaa";
var reg = /a/;
str = str.replace( reg , "!");
console.log(str); // !gghdsaa
var reg = /a/g;
str = str.replace( reg , "!");
console.log(str); //!gghds!!
//example ---- 敏感词过滤
str = "Funk, what the funk, son of bitch";
console.log( "过滤前 : " + str); //过滤前 : Funk, what the funk, son of bitch
reg = /funk|bitch+/ig;
str = str.replace( reg , function(string , second){
  'use strict';
  //callback的第一个形参就是匹配成功的字符
  var result = "" , i = 0;
  while ( i < string.length ) {
    result += "*";
    i++;
  }
  return result;
});
console.log( "过滤后 : " + str); //过滤后 : ****, what the ****, son of *****

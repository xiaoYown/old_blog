常用转义字符

. : 任意字符
\.: 真正的.
\s: 空格
\S: 非空格
\d: 数字
\D: 非数字
\w: 字符 (包括 字母 ， 数字 ， 下划线 _)
\W: 非字符
\b: 独立的部分 ( 起始 , 结束 , 空格 ) ----example 1
\B: 非独立的部分
\1: 重复第1个子项 ---- example 2
\2: 重复第2个子项

example 1 ---- 获取class的方法

function getByClass( parent , classe ){
  var arr = [];
  var eles = parent.getElementsByTagName('*');
  var len = eles.length;
  //当正则需要传参的时候，一定要用全称的写法
  var reg = new RegExp( '\\b' + classe + '\\b');
  for(var i = 0 ; i < len ; i++){
    if( reg.test( eles[i].className ) ) {
      arr.push( eles[i] );
    }
  }
  return arr;
}
var box01s = getByClass( document , "box01" );
var len = box01s.length;
for(var i = 0 ; i < len ; i++) {
  console.log(box01s[i])
  box01s[i].style.backgroundColor = "#db214c";
}

example 2 ---- 重复的子项

var str = "aadadhhjmjjjjmjdgggghhmmmm";
var arr = str.split("");
str = arr.sort().join("");
var value = [];
var index = 0;
var reg = /(\w)\1+/g;
str.replace( reg , function($0){
  if ( index < $0.length ) {
    index = $0.length;
  }
});
str.replace( reg , function($0,$1){
  if ( index === $0.length ) {
    value.push($1);
  }
});
var len = value.length;
console.log( "出现次数最多的为" + index + "次");
for( var i = 0 ; i < len ; i ++) {
  console.log(value[i]);
}
